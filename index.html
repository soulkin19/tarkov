// --- 以下、修正・追加した主要な関数・セクション --- // 1. セカンダリ武装の使用 (威力の変更と検知距離のパラメータ設定) function useDefense() { if(player.defenseCD > 0) return; const tAngle = player.turretAngle; if(player.defense === 1) { // AT4 const s = { x: player.x, y: player.y, vx: Math.cos(tAngle)*6, vy: Math.sin(tAngle)*6, power: 30, // 威力を30に変更 color: "#c0392b", isMissile: true, isExplosive: true, owner: player.id, ownerName: player.name, weaponName: "AT4", life: 500 }; shells.push(s); socket?.send(JSON.stringify({toH: HUB_NAME, msg: JSON.stringify({type:'fire', shell:s})})); player.defenseCD = 500; } else if(player.defense === 6) { // ADS player.adsActive = 240; player.defenseCD = 700; } else if(player.defense === 7) { // C4 if (!player.c4) { player.c4 = { x: player.x, y: player.y }; socket?.send(JSON.stringify({toH: HUB_NAME, msg: JSON.stringify({type:'c4_placed', x: player.x, y: player.y, ownerId: player.id})})); } else { createExplosion(player.c4.x, player.c4.y, 60, player.name, "リモートC4", true); player.c4 = null; player.defenseCD = 400; } } else if(player.defense === 4) { // 近接信管誘導弾 const s = { x: player.x, y: player.y, vx: Math.cos(tAngle)*14, vy: Math.sin(tAngle)*14, power: 28, // ダメージを増加 color: "#2ecc71", isProxy: true, // 検知フラグ proxDist: 100, // 検知距離を延長 isExplosive: true, owner: player.id, ownerName: player.name, weaponName: "近接信管弾", life: 250 }; shells.push(s); socket?.send(JSON.stringify({toH: HUB_NAME, msg: JSON.stringify({type:'fire', shell:s})})); player.defenseCD = 350; } } // 2. アップデート処理 (近接信管の爆発ロジック追加) function update() { // ... (移動処理などはそのまま) ... for (let i = shells.length-1; i >= 0; i--) { const s = shells[i]; // 近接信管 (isProxy) のロジック if (s.isProxy) { let triggered = false; // 敵プレイヤーとの距離を計算 for (let id in players) { if (s.owner !== id && Math.hypot(players[id].x - s.x, players[id].y - s.y) < s.proxDist) { triggered = true; break; } } // 自分が敵から撃たれた場合も検知 if (s.owner !== player.id && Math.hypot(player.x - s.x, player.y - s.y) < s.proxDist) { triggered = true; } if (triggered) { createExplosion(s.x, s.y, s.power, s.ownerName, s.weaponName, s.owner === player.id); shells.splice(i, 1); continue; } } // --- ADSの防御判定 --- if (player.adsActive > 0 && s.owner !== player.id && Math.hypot(player.x - s.x, player.y - s.y) < 120) { explosions.push({ x: s.x, y: s.y, r: 0, life: 10, critical: false }); shells.splice(i,1); continue; } // ... (通常の壁衝突・直撃判定) ... } // --- 通信データに ADS の状態を乗せる --- if (socket?.readyState === 1) { socket.send(JSON.stringify({toH: HUB_NAME, msg: JSON.stringify({ type: 'move', id: player.id, x: player.x, y: player.y, angle: player.angle, tAngle: player.turretAngle, hp: player.hp, name: player.name, tankType: player.type, adsActive: player.adsActive // これを追加 })})); } } // 3. 描画処理 (他人のADSも見えるように) const drawT = (t, col) => { const spec = tankSpecs[t.tankType || 'medium'], s = spec.size; ctx.save(); ctx.translate(t.x, t.y); // ADSの描画条件を「t.adsActive > 0」に変更（自分・他人共通） if (t.adsActive > 0) { ctx.strokeStyle = "rgba(0, 255, 255, 0.6)"; ctx.lineWidth = 3; ctx.setLineDash([5, 5]); // 点線にして「起動中」感を出す ctx.beginPath(); ctx.arc(0, 0, 120, 0, Math.PI*2); ctx.stroke(); ctx.setLineDash([]); // 元に戻す } // ... (戦車の描画) ... ctx.restore(); };
